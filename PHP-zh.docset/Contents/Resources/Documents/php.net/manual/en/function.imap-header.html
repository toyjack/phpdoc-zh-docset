Here is a clean way to decode encoded imap headers that will work in all cases :

function fix_text($str)
{
    $subject = '';
    $subject_array = imap_mime_header_decode($str);

    foreach ($subject_array AS $obj)
        $subject .= rtrim($obj->text, "\t");

    return $subject;
}
