Bookmarks as URL

<?php
  $action = $p->create_action("URI", "url=http://www.example.com");
  $p->create_bookmark("www.example.com", "action {activate $action}");
?>

    OR

<?php
  $action = pdf_create_action($p, "URI", "url=http://www.example.com");
  pdf_create_bookmark($p, "www.example.com", "action {activate $action}");
?>
