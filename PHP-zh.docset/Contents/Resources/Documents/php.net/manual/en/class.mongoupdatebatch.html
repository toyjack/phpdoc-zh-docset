    Complete example:

    <?php
    $mc     = new MongoClient('localhost');
    $collection = $mc->selectCollection('blog', 'users');

    $update = array(
          'q' => array('foo' => 'bar'),
          'u' => array('$set' => array('foo' => 'baz')),
          'multi' => false,
          'upsert' => false,
    );
    $batch = new MongoUpdateBatch($collection);
    $batch->add((object) $update);
    $batch->execute();
